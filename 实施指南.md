```typescript
import React, {createContext, useContext, useState} from 'react';

type ThemeMode = 'professional' | 'creative';

const ThemeContext = createContext<{
  mode: ThemeMode;
  setMode: (mode: ThemeMode) => void;
}>({
  mode: 'professional',
  setMode: () => {},
});

export function ThemeProvider({children}: {children: React.ReactNode}) {
  const [mode, setMode] = useState<ThemeMode>('professional');
  
  return (
    <ThemeContext.Provider value={{mode, setMode}}>
      <div data-theme={mode}>
        {children}
      </div>
    </ThemeContext.Provider>
  );
}

export const useTheme = () => useContext(ThemeContext);
```

```typescript
// apps/web/src/app/hooks/useFormValidation.ts
import {useState, useEffect} from 'react';

export function useFormValidation(content: string, taskName: string) {
  const [errors, setErrors] = useState<Record<string, string>>({});
  const [isValid, setIsValid] = useState(false);
  
  useEffect(() => {
    const newErrors: Record<string, string> = {};
    
    if (content.trim().length === 0) {
      newErrors.content = '请输入题目内容';
    } else if (content.length > 500) {
      newErrors.content = '题目内容不能超过500字';
    }
    
    if (taskName.trim().length === 0) {
      newErrors.taskName = '请输入任务名称';
    } else if (taskName.length > 50) {
      newErrors.taskName = '任务名称不能超过50字';
    }
    
    setErrors(newErrors);
    setIsValid(Object.keys(newErrors).length === 0);
  }, [content, taskName]);
  
  return {errors, isValid};
}
```

```typescript
'use client';

import React from 'react';
import {createButtonStyle} from '../ui-primitives-v2';

export function LoadingButton({
  children,
  isLoading,
  disabled,
  onClick,
  variant = 'primary',
  size = 'md',
}: {
  children: React.ReactNode;
  isLoading?: boolean;
  disabled?: boolean;
  onClick?: () => void;
  variant?: 'primary' | 'secondary' | 'outline';
  size?: 'sm' | 'md' | 'lg';
}) {
  return (
    <button
      onClick={onClick}
      disabled={disabled || isLoading}
      style={{
        ...createButtonStyle(variant, size),
        position: 'relative',
        ...(isLoading || disabled ? {opacity: 0.6, cursor: 'not-allowed'} : {}),
      }}
    >
      {isLoading && (
        <span style={spinnerStyle}>
          <Spinner />
        </span>
      )}
      <span style={isLoading ? {visibility: 'hidden'} : {}}>
        {children}
      </span>
    </button>
  );
}

function Spinner() {
  return (
    <svg
      width="20"
      height="20"
      viewBox="0 0 24 24"
      fill="none"
      stroke="currentColor"
      strokeWidth="2"
      style={{animation: 'spin 1s linear infinite'}}
    >
      <circle cx="12" cy="12" r="10" opacity="0.25" />
      <path d="M12 2a10 10 0 0 1 10 10" opacity="0.75" />
    </svg>
  );
}

const spinnerStyle: React.CSSProperties = {
  position: 'absolute',
  left: '50%',
  top: '50%',
```

```typescript
// apps/web/src/app/hooks/useMediaQuery.ts
import {useState, useEffect} from 'react';

export function useMediaQuery(query: string) {
  const [matches, setMatches] = useState(false);
  
  useEffect(() => {
    const media = window.matchMedia(query);
    setMatches(media.matches);
    
    const listener = (e: MediaQueryListEvent) => setMatches(e.matches);
    media.addEventListener('change', listener);
    
    return () => media.removeEventListener('change', listener);
  }, [query]);
  
  return matches;
}

// 使用示例
export function useBreakpoint() {
  const isMobile = useMediaQuery('(max-width: 768px)');
  const isTablet = useMediaQuery('(min-width: 769px) and (max-width: 1024px)');
  const isDesktop = useMediaQuery('(min-width: 1025px)');
  
  return {isMobile, isTablet, isDesktop};
}
```

```typescript
// apps/web/src/app/hooks/useJobStatus.ts
import {useState, useEffect} from 'react';
import {fetchJobStatus} from '../lib/api-client';

export function useJobStatus(jobId: string, interval = 2000) {
  const [status, setStatus] = useState<JobStatus | null>(null);
  const [error, setError] = useState<Error | null>(null);
  
  useEffect(() => {
    let active = true;
    let timeoutId: NodeJS.Timeout;
    
    const poll = async () => {
      try {
        const latest = await fetchJobStatus(jobId);
        
        if (active) {
          setStatus(latest);
          
          // 如果任务完成或失败,停止轮询
          if (latest.status === 'completed' || latest.status === 'failed') {
            return;
          }
          
          // 继续轮询
          timeoutId = setTimeout(poll, interval);
        }
      } catch (err) {
        if (active) {
          setError(err as Error);
        }
      }
    };
    
    poll();
    
    return () => {
      active = false;
      clearTimeout(timeoutId);
    };
  }, [jobId, interval]);
  
  return {status, error};
}
```

```typescript
// apps/web/src/app/hooks/useAutoVoicePreview.ts
import {useState, useEffect} from 'react';
import {previewTts} from '../lib/api-client';
import {recommendVoicePreset} from '../voice-recommendation';

export function useAutoVoicePreview(
  content: string,
  style: string,
  targetDurationSec: number
) {
  const [preview, setPreview] = useState<VoicePreview | null>(null);
  const [isLoading, setIsLoading] = useState(false);
  
  useEffect(() => {
    // 只有当内容足够长时才自动预览
    if (content.length < 10) {
      setPreview(null);
      return;
    }
    
    let active = true;
    
    const generatePreview = async () => {
      setIsLoading(true);
      
      try {
        const recommendation = recommendVoicePreset({
          content,
          style,
          targetDurationSec,
        });
        
        const result = await previewTts({
          text: content,
          style,
          targetDurationSec,
          voice: recommendation.voice,
          speechRate: recommendation.speechRate,
        });
        
        if (active) {
          setPreview(result);
        }
      } catch (error) {
        console.error('Auto preview failed:', error);
      } finally {
        if (active) {
          setIsLoading(false);
        }
      }
    };
    
    // 防抖: 用户停止输入500ms后才生成预览
    const timeoutId = setTimeout(generatePreview, 500);
    
    return () => {
      active = false;
      clearTimeout(timeoutId);
    };
  }, [content, style, targetDurationSec]);
  
  return {preview, isLoading};
}
```

```typescript
const CACHE_TTL = 5 * 60 * 1000; // 5分钟

export async function listJobsWithCache() {
  const cacheKey = 'jobs-list';
  const cached = cache.get(cacheKey);
  
  if (cached && Date.now() - cached.timestamp < CACHE_TTL) {
    return cached.data;

  return data;
}
```

```typescript
// apps/web/src/app/components/JobBatchActions.tsx
'use client';

import {useState} from 'react';
import {createButtonStyle} from '../ui-primitives-v2';

export function JobBatchActions({jobs}: {jobs: Job[]}) {
  const [selectedIds, setSelectedIds] = useState<Set<string>>(new Set());
  
  const toggleSelection = (jobId: string) => {
    const newSelection = new Set(selectedIds);
    if (newSelection.has(jobId)) {
      newSelection.delete(jobId);
    } else {
      newSelection.add(jobId);
    }
    setSelectedIds(newSelection);
  };
  
  const selectAll = () => {
    setSelectedIds(new Set(jobs.map(j => j.jobId)));
  };
  
  const clearSelection = () => {
    setSelectedIds(new Set());
  };
  
  const handleBatchDelete = async () => {
    if (!confirm(`确定要删除 ${selectedIds.size} 个任务吗?`)) {
      return;
    }
    
    await Promise.all(
      Array.from(selectedIds).map(id => deleteJob(id))
    );
    
    clearSelection();
  };
  
  return (
    <div style={batchActionsStyle}>
      {selectedIds.size > 0 && (
        <>
          <span>已选择 {selectedIds.size} 个任务</span>
          <button
            onClick={handleBatchDelete}
            style={createButtonStyle('danger', 'sm')}
          >
            批量删除
          </button>
          <button
            onClick={clearSelection}
            style={createButtonStyle('outline', 'sm')}
          >
            取消选择
          </button>
        </>
      )}
      
      {selectedIds.size === 0 && (
        <button
          onClick={selectAll}
          style={createButtonStyle('outline', 'sm')}
        >
          全选
        </button>
      )}
    </div>
  );
}

const batchActionsStyle: React.CSSProperties = {
  display: 'flex',
  gap: '12px',
  alignItems: 'center',
  padding: '12px',
  background: '#F5F5F4',
```

## 💡 常见问题

### Q: 是否需要完全替换现有的 UI?

A: 不需要。建议采用渐进式迁移策略:
1. 先引入新的设计系统
2. 新功能使用新样式
3. 逐步重构现有页面
4. 保持两套样式共存一段时间

### Q: 如何处理用户习惯了旧界面?

A: 提供主题切换功能,让用户可以选择"专业模式"或"创意模式"。

### Q: 性能优化会影响功能吗?

A: 不会。代码分割和懒加载只是改变了加载时机,不影响功能完整性。

### Q: 移动端适配工作量大吗?

A: 如果使用响应式设计系统,工作量可控。建议优先适配核心流程(创建视频、查看任务)。
